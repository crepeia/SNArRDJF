##Plotting Diameter
##Plotting Betweenness Centrality - (Vertex)

```r
set.seed(123)
d <- get.diameter(x2, weights =E(x2)$x2 )
E(x2)$color_d <- "grey"
E(x2)$width_d <- 0.1
E(x2, path=d)$color_d <- "red"
E(x2, path=d)$width_d <- 2
V(x2)$label.color_d <- "blue"
V(x2)$color_d  <- "SkyBlue2"
V(x2)$color_label  <- "Non diameter"
V(x2)[d]$label.color_d <- "black"
V(x2)[d]$color_d <- "red"
V(x2)[d]$color_label <- "Diameter"

#Plotting based only on degree measures 
edge.start <- ends(x2, es=E(x2), names=F)[,1]

# Fixing ego
minC <- rep(-Inf, vcount(x2))
maxC <- rep(Inf, vcount(x2))
minC[1] <- maxC[1] <- 0
co <- layout_with_fr(x2, niter=10^4, minx=minC, maxx=maxC,miny=minC, maxy=maxC, weights = E(x2)$x2)

#Plotting
plot(x2, 
     layout=co,
     edge.color=E(x2)$color_d,
     edge.arrow.size=(betweenness(x2, weights = E(x2)$x2)+1)/100000,
     edge.width=E(x2)$width_d,
     edge.curved = TRUE,
     vertex.color=V(x2)$color_d,
     vertex.size=degree(x2)/5,
     vertex.frame.color="black",
     vertex.label.color=V(x2)$label.color_d,
     vertex.label=get.vertex.attribute(x2,"LABEL_COR"),
     vertex.label.cex=(betweenness(x2, weights = E(x2)$x2)+1)/10000,
     vertex.label.dist=0,
     rescale=F,
     xlim=range(co[,1]), 
     ylim=range(co[,2])
     )
axis(1)
axis(2)

#Solving Problems with legend rendering 
a<-V(x2)$color_label 
b<-V(x2)$color_d
c<-table(a,b)
d<-as.data.frame(c)
e<-subset(d, d$Freq>0)
e<-e[order(e$a,decreasing=T),] 
f<-t(e$a)
g<-t(e$b)

#Adding Legend
legend(x=range(co[,1])[2], 
       y=range(co[,2])[2],
       legend=as.character(f),
       pch=21,
       col = "#777777", 
       pt.bg=as.character(g),
       pt.cex=2,
       bty="n", 
       ncol=1,
       lty=1,
       cex = .3)

#Adding Title
  title("Diameter of 1.2 - ALGUM RELACIONAMENTO (x2)", sub = "Source: from authors ")
  text( 
    x=range(co[,1])[1],
    y=range(co[,2])[1], 
      labels =    sprintf("Diameter: %.2f",diameter(x2, weights = E(x2)$x2))
             )
```
